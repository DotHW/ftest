import 'package:flutter/material.dart';

void main() {
  runApp(MaterialApp(home: HomePage()));
}

class HomePage extends StatefulWidget {
  @override
  State createState() => _HomePageState();
}

class _HomePageState extends State {
  bool change = false;

    return Scaffold(
      appBar: AppBar(
        title: Text("Flutter Animated App"),
      ),
      body: Center(
        child: GestureDetector(
          onTap: () {

              change = !change;
            });
          },

            duration: Duration(milliseconds: 500),
            width: change ? 300 : 200,
            height: change ? 300 : 200,
            child: Image.network(
              "https://picsum.photos/200",
              fit: BoxFit.cover,
            ),
          ),
        ),
      ),
    );

      timer?.cancel();

        ),

          decoration: BoxDecoration(
            color: isRunning ? Colors.deepPurple : Colors.teal,
            borderRadius: BorderRadius.circular(isRunning ? 30 : 5:30),
          ),
          child: const Icon(
            Icons.timer,
            color: Colors.white,
            size: 60,
          ),
        ),

        const SizedBox(height: 30),

        Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            ElevatedButton(
              onPressed: startTimer,
              child: const Text("Start"),
