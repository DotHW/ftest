import 'package:flutter/material.dart';
import 'dart:math';

void main() {
  runApp(MyApp());
}

class MyApp extends StatefulWidget {
  const MyApp({super.key});

  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  bool f = false;
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: Scaffold(
        body: Container(
          child: GestureDetector(
            onTap: () {
              setState(() {
                f = !f;
              });
            },
            child: AnimatedContainer(
              duration: Duration(seconds: 3),
              color: Colors.green,
              width:  500,
              height:  500,
              child: AnimatedContainer(
                  duration: Duration(seconds: 3),
                width:  300,
                height:  300,
                child: Image(
                  image: NetworkImage("http://picsum.photos/200")
                  ),

              @override
              Widget build(BuildContext context) {
              return MaterialApp(
              home: Scaffold(
              body: Container(
              child: GestureDetector(
              onTap: () {
              setState(() {
              f = !f;
              });
              },

            ),
          ),




            class _MyAppState extends State<MyApp> with SingleTickerProviderStateMixin {
            late AnimationController c;
            late Animation<double> a;

            @override
            void initState() {
            super.initState();

            c = AnimationController(duration: Duration(seconds: 10), vsync: this);

            a = Tween<double>(begin: 1, end: 100).animate(c);

            c.repeat();
            }

            @override
            Widget build(BuildContext context) {
            return MaterialApp(
            home: Scaffold(
            body: Container(
            child: Column(
            children: [
            AnimatedBuilder(
            animation: a,
            builder: (context, child) {
            return Text(
            "${a.value.toInt()}",
            style: TextStyle(fontSize: 30),
            );
            },
            ),
            ),
            ElevatedButton(
            onPressed: () {
            setState(() {
            c.repeat();
            });
            },
            child: Text("start"),
            ),
            ElevatedButton(
            onPressed: () {
            setState(() {
            c.stop();
            });
            },
            child: Text("Stop"),
            ),



            ),





            );


  }
}
