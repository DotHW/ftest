import 'dart:convert';
import 'package:http/http.dart' as http;
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home:Home(),
    );
  }
}

class Home extends StatefulWidget {
  const Home({super.key});

  @override
  State<Home> createState() => _HomeState();
}

class _HomeState extends State<Home> {
  List name = [];
  List domains = [];
  List web_pages = [];

  Future <void> fetchName() async{
    var url = Uri.parse("http://unviversities.hoplabs.com/search?country=jordan");
    var response = await http.get(url);
    if(response.statusCode == 200) {
      var mydata = jsonDecode(response.body);
      setState(() {
        name = mydata["name"];
      });
    }
  }

  Future <void> fetchDomains() async{
    var url = Uri.parse("http://unviversities.hoplabs.com/search?country=jordan");
    var response = await http.get(url);
    if(response.statusCode == 200) {
      var mydata = jsonDecode(response.body);
      setState(() {
        domains = mydata["domains"];
      });
    }
  }

  Future <void> fetchWebPages() async{
    var url = Uri.parse("http://unviversities.hoplabs.com/search?country=jordan");
    var response = await http.get(url);
    if(response.statusCode == 200) {
      var mydata = jsonDecode(response.body);
      setState(() {
        web_pages = mydata["web_pages"];
      });
    }
  }

  @override
  void initState() {
    super.initState();
    fetchName();
    fetchDomains();
    fetchWebPages();
  }

 

  @override
  Widget build(BuildContext context) {
    return Scaffold(
        appBar: AppBar(
          title: Text("Latest News"),
          centerTitle: true,
          backgroundColor: Colors.amberAccent,
        ),
        body: Padding(
          padding: const EdgeInsets.all(8.0),
          child: SafeArea(
            child: ListView.builder(



                itemCount: web_pages.length,
                itemBuilder: (context,index){
                  return Card(
                    elevation: 5,
                    child: Column(
                      children: [
                        ClipRRect(
                          borderRadius: BorderRadiusGeometry.only(
                            topLeft: Radius.circular(8),
                            topRight: Radius.circular(8),
                          ),
                          child: Image.network(name[index][""],
                            fit: BoxFit.cover,
                            height: 100,
                          ),
                        ),
                        SizedBox(height: 10,),
                        Text(domains[index][""]),
                        Padding(
                          padding: const EdgeInsets.all(8.0),
                          child: Text(web_pages[index][""],
                            style: TextStyle(fontWeight: FontWeight.bold),
                          ),
                        ),
                      ],
                    ),

                  );
                }
            ),
          ),
        )
    );
  }
}
