import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:http/http.dart' as http;

import 'DetailScreen.dart';

class DisplayScreen extends StatefulWidget {
  const DisplayScreen({super.key});

  @override
  State<DisplayScreen> createState() => _DisplayScreenState();
}

class _DisplayScreenState extends State<DisplayScreen> {
  List results = [];
  bool isLoading = false;
  String query = '';

  Future<void> display(String name) async {
    if (name.isEmpty) {
      setState(() => results = []);
      return;
    }

    setState(() => isLoading = true);

    var url = Uri.parse("https://universities.hipolabs.com/search?country=jordan");
    var response = await http.get(url);

    if (response.statusCode == 200) {
      var data = jsonDecode(response.body);
      setState(() {
        results = data;
        isLoading = false;
      });
    } else {
      setState(() => isLoading = false);
    }
  }

  String _stripHtml(String html) {
    return html.replaceAll(RegExp(r'<[^>]*>'), '');
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        backgroundColor: const Color(0xFF141414),
        title: const Text(
          'Jordan Universities',
          style: TextStyle(
            color: Colors.white,
            fontWeight: FontWeight.bold,
            fontSize: 22,
          ),
        ),
      ),
      body: Center(
        child:
           Column(
             children: [
               ListView.builder(
                       padding: const EdgeInsets.all(12),
                       itemCount: results.length,
                       itemBuilder: (context, index) {
                         var show = results[index]['show'];
                         var name = show['name'] ?? 'No Name';
                         var imageUrl = show['image']?['medium'];
                         var domains = show['domains']?['average'];
                         var web_pages = (show['web_pages'] as List?)?.join(', ') ?? '';


                         return GestureDetector(
                onTap: () {
                  Navigator.push(
                    context,
                    MaterialPageRoute(
                      builder: (context) => DetailScreen(show: show),
                    ),
                  );
                },
                child: Container(
                  margin: const EdgeInsets.only(bottom: 12),
                  decoration: BoxDecoration(
                    color: const Color(0xFF1E1E1E),
                    borderRadius: BorderRadius.circular(12),
                  ),
                  child: Row(
                    children: [
                      // Image
                      ClipRRect(
                        borderRadius: const BorderRadius.only(
                          topLeft: Radius.circular(12),
                          bottomLeft: Radius.circular(12),
                        ),
                        child: imageUrl != null
                            ? Image.network(
                          imageUrl,
                          width: 90,
                          height: 120,
                          fit: BoxFit.cover,
                          errorBuilder: (_, __, ___) => _placeholder(),
                        )
                            : _placeholder(),
                      ),
                      const SizedBox(width: 12),
                      // Info
                      Expanded(
                        child: Padding(
                          padding: const EdgeInsets.symmetric(vertical: 12, horizontal: 4),
                          child: Column(
                            crossAxisAlignment: CrossAxisAlignment.start,
                            children: [
                              Text(
                                name,
                                style: const TextStyle(
                                  color: Colors.white,
                                  fontWeight: FontWeight.bold,
                                  fontSize: 16,
                                ),
                              ),
                              if (web_pages.isNotEmpty) ...[
                                const SizedBox(height: 4),
                                Text(
                                  web_pages,
                                  style: const TextStyle(
                                    color: Color(0xFFE50914),
                                    fontSize: 12,
                                  ),
                                ),
                              ],
                              const SizedBox(height: 6),

                              const SizedBox(height: 8),
                              if (domains != null)
                                Row(
                                  children: [
                                    const Icon(Icons.star, color: Colors.amber, size: 16),
                                    const SizedBox(width: 4),
                                    Text(
                                      domains ,
                                      style: const TextStyle(
                                        color: Colors.amber,
                                        fontWeight: FontWeight.bold,
                                        fontSize: 13,
                                      ),
                                    ),
                                  ],
                                ),
                            ],
                          ),
                        ),
                      ),
                      const Padding(
                        padding: EdgeInsets.only(right: 12),
                        child: Icon(Icons.arrow_forward_ios, color: Colors.white24, size: 16),
                      ),
                    ],
                  ),
                ),
                         );
                       },
                     ),
             ],
           ),
      ) );
  }

  Widget _placeholder() {
    return Container(
      width: 90,
      height: 120,
      color: const Color(0xFF2A2A2A),
      child: const Icon(Icons.image_not_supported, color: Colors.white24),
    );
  }
}
