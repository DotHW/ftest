import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(home: Home());
  }
}

class Home extends StatefulWidget {
  const Home({super.key});

  @override
  State<Home> createState() => _HomeState();
}

class _HomeState extends State<Home> {
  final String imgURL = 'https://picsum.photos/200';
  int oContainerS = 500;
  int iContainerS = 300;
  int imgS = 200;
  bool isClicked = false;
  int timer = 0;
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(title: Text('Flutter Animated App')),
      body: Column(
        children: [
          Center(
            child: AnimatedContainer(
              duration: Duration(seconds: 3),
              width: isClicked ? 800 : 500,
              height: isClicked ? 800 : 500,
              decoration: BoxDecoration(color: Colors.red),
              child: AnimatedContainer(
                duration: Duration(seconds: 3),
                width: isClicked ? 600 : 300,
                height: isClicked ? 600 : 300,
                decoration: BoxDecoration(color: Colors.blue),
                child: AnimatedContainer(
                  duration: Duration(seconds: 3),
                  width: isClicked ? 400 : 200,
                  height: isClicked ? 400 : 200,
                  child: Image.network(imgURL, fit: BoxFit.scaleDown),
                ),
              ),
            ),
          ),
          TextButton(
            onPressed: () {
              isClicked = !isClicked;
            },
            child: Text('animate'),
          ),
          SizedBox(height: 20),
          TextFormField(),
          Container(
            decoration: BoxDecoration(color: Colors.green),
            child: Text(timer.toString()),
          ),
          Center(
            child: Row(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                TextButton(onPressed: () {}, child: Text('Start')),
                SizedBox(width: 10),
                TextButton(onPressed: () {}, child: Text('Stop')),
              ],
            ),
          ),
        ],
      ),
    );
  }
}
